class Solution:
    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:
        # Two pointers
        # nlogn + n + m

        res = []
        products.sort()

        l, r = 0, len(products)-1
        for i in range(len(searchWord)):
            c = searchWord[i]

            # l å’Œ r æŒ‡å‘å•è¯
            # products[l][i] each character in the product
            while l <= r and (len(products[l]) <= i or products[l][i] != c): # Don't have more character or the ith character is not what we are looking for
                l += 1
            while l <= r and (len(products[l]) <= i or products[r][i] != c):
                r -= 1

            res.append([])
            remain = r - l + 1
            for j in range(min(3, remain)):
                res[-1].append(products[l + j]) # æœ€å[]é‡Œappend products,ä¸€ç›´åŠ  3 ä¸ªï¼Œå¦‚æœremain çš„å°äº 3 ä¸ªï¼Œå› ä¸ºå–äº† minï¼Œæ‰€ä»¥ğŸ‰‘
